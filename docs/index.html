    Programs

    How to use programs? check it.
    Example1:- echo \stMp\stMp::_isEmpty(''); //Output:- 0
    Example2:- echo \stMp\stMp::_getStrLen('Demo'); //Output:- 4
    Example3:- echo \stMp\stMp::_trim('Demo'); //Output:- Demo
    Example4:- echo \stMp\stMp::_ltrim('Demo'); //Output:- Demo
    Example5:- echo \stMp\stMp::_rtrim('Demo'); //Output:- Demo
    Example6:- echo \stMp\stMp::_stripSpace('De mo'); //Output:- 4
    Example7:- echo \stMp\stMp::_lower('DEMO'); //Output:- demo
    Example8:- echo \stMp\stMp::_upper('demo'); //Output:- DEMO
    Example9:- echo \stMp\stMp::_stripQuotes('"Demo"'); //Output:- Demo
    Example10:- echo \stMp\stMp::_replace('Demo','DEMO); //Output:- Demo
    Example11:- echo \stMp\stMp::_upperCamelCase('Demo'); //Output:- Demo
    Example12:- echo \stMp\stMp::_slug('Demo Demo'); //Output:- Demo-Demo
    Example13:- echo \stMp\stMp::_snakeCase('Demo Demo'); //Output:- Demo_Demo
    Example14:- echo \stMp\stMp::_dotNotation('Demo Demo'); //Output:- Demo.Demo
    Example15:- echo \stMp\stMp::_isPalindrome('Ana'); //Output:- 
    Example16:- echo \stMp\stMp::_wordCount('Demo'); //Output:- 1
    Example17:- echo \stMp\stMp::_htmlRemoveTags('Demo'); //Output:- Demo
    Example18:- echo \stMp\stMp::_removeSlashes('/Demo/'); //Output:- Demo
    Example19:- echo \stMp\stMp::_removeDoubleSlashes('//Demo//'); //Output:- Demo
    Example20:- echo \stMp\stMp::_reduceMultiples('Demo,,Demo'); //Output:- Demo,Demo
    Example21:- echo \stMp\stMp::_arrayToString('Demo'); //Output:- Demo
    Example22:- echo \stMp\stMp::_reverseString('Demo'); //Output:- omeD
    Example23:- echo \stMp\stMp::_randomString(6); //Output:- saerth
    Example24:- echo \stMp\stMp::_ucFirst("demo"); //Output:- Demo
    Example25:- echo \stMp\stMp::_ucLirst("demo"); //Output:- emO
    Example26:- echo \stMp\stMp::_isAnagram("demo"); //Output:- 0
    Example27:- echo \stMp\stMp::_countSentence("This is. This is.); //Output:- 2
    Example28:- echo \stMp\stMp::_removeDuplicateChars("This is. This is.); //Output:- This is.
    Example29:- echo \stMp\stMp::_removeComment("/*This is.*/); //Output:- This is.
    Example30:- echo \stMp\stMp::_stripDigit("9This"); //Output:- This
    Example31:- echo \stMp\nodiacritic\Nodiacritic::_removeDiacritic("≈•hi≈õ"); //Output:- this
